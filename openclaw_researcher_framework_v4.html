职责划分：Python 脚本只做计算，不做编排。Claude API（Haiku/Sonnet）在脚本内 直接调用，Voyage Embedding 也在此处。脚本通过 stdout 返回结果，Agent 读取即可。 编排逻辑全部由 AGENTS.md + SKILL.md 定义。
workspace/scripts/fetch_papers.pyArXiv · Haiku query生成 · Sonnet 信息抽取
import arxiv, json, argparse
from anthropic import Anthropic

client = Anthropic()  # 读取 ANTHROPIC_API_KEY

def gen_queries(task: dict) -> list:
    """Haiku：将研究方向分解为 3-5 个 ArXiv 检索词组合（~$0.001）"""
    r = client.messages.create(
        model="claude-haiku-4-5-20251001", max_tokens=512,
        system='只返回JSON，格式: {"queries":["q1","q2",...]}',
        messages=[{"role": "user", "content":
            f"ASR研究方向拆解为3-5个ArXiv关键词组合:\n{task['tech_direction']}"}]
    )
    return json.loads(r.content[0].text)["queries"]

def extract_info(title: str, abstract: str) -> dict:
    """Sonnet：抽取方法/数据集/指标，并打相关性分（0-10）"""
    r = client.messages.create(
        model="claude-sonnet-4-6", max_tokens=512,
        system='返回JSON: {"methods":[],"datasets":[],"metrics":{},"relevance":0}',
        messages=[{"role": "user", "content":
            f"Title: {title}\nAbstract: {abstract[:800]}"}]
    )
    return json.loads(r.content[0].text)

def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--task"); ap.add_argument("--output")
    args = ap.parse_args()

    task    = json.load(open(args.task))
    queries = gen_queries(task)
    print(f"🔍 生成 {len(queries)} 个检索 query")

    papers = []
    for q in queries:
        results = list(arxiv.Search(query=q, max_results=8,
            sort_by=arxiv.SortCriterion.Relevance).results())
        for p in results:
            info = extract_info(p.title, p.summary)
            if info.get("relevance", 0) >= 6:  # 过滤低相关
                papers.append({
                    "id": p.entry_id, "title": p.title,
                    "abstract": p.summary, "url": p.pdf_url,
                    "info": info
                })

    json.dump(papers, open(args.output, "w"), ensure_ascii=False, indent=2)
    print(f"✓ 高相关论文 {len(papers)} 篇 → {args.output}")

if __name__ == "__main__": main()
scripts/embed_papers.pyVoyage AI · ChromaDB
import voyageai, chromadb
import json, argparse
from pathlib import Path

vo    = voyageai.Client()   # VOYAGE_API_KEY
chroma = chromadb.PersistentClient(
    path=str(Path("~/.openclaw/workspace/data/chroma"
    ).expanduser()))
coll  = chroma.get_or_create_collection("papers")

def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--papers")
    args = ap.parse_args()

    papers = json.load(open(args.papers))
    texts  = [p["abstract"][:2000] for p in papers]

    # voyage-3.5，document 类型，batch=64
    embeddings = []
    for i in range(0, len(texts), 64):
        res = vo.embed(texts[i:i+64],
            model="voyage-3.5",
            input_type="document")
        embeddings.extend(res.embeddings)

    coll.upsert(
        ids       =[p["id"] for p in papers],
        embeddings=embeddings,
        documents =texts,
        metadatas =[{"title": p["title"],
                     "url":   p.get("url", "")}
                    for p in papers]
    )
    print(f"✓ 向量化 {len(papers)} 篇 → ChromaDB")

if __name__ == "__main__": main()

import json, argparse, voyageai, chromadb
from anthropic import Anthropic
from pathlib import Path

client = Anthropic()
vo     = voyageai.Client()
coll   = chromadb.PersistentClient(
  path=str(Path("~/.openclaw/workspace/data/chroma"
  ).expanduser())
).get_collection("papers")

def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--task")
    ap.add_argument("--papers")
    ap.add_argument("--output")
    args = ap.parse_args()

    task = json.load(open(args.task))

    # RAG：用研究方向 query 检索最相关论文
    q_embed = vo.embed([task["tech_direction"]],
        model="voyage-3.5",
        input_type="query").embeddings[0]
    top_k = coll.query(
        query_embeddings=[q_embed], n_results=12)

    # Sonnet 生成 3000 字综述
    r = client.messages.create(
        model="claude-sonnet-4-6",
        max_tokens=3000,
        system="""你是资深 ASR 研究员，写技术综述：

import subprocess, json, yaml, sys, time, mlflow, argparse
from anthropic import Anthropic
from pathlib import Path

client = Anthropic()

def summarize_error(stderr: str) -> dict:
    """Haiku：解析 stderr，提取错误类型和修复建议（~$0.001）"""
    r = client.messages.create(
        model="claude-haiku-4-5-20251001", max_tokens=256,
        system='返回JSON: {"type":"...","line":"...","fix":"..."}',
        messages=[{"role":"user", "content": stderr[-3000:]}]
    )
    return json.loads(r.content[0].text)

def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--config"); ap.add_argument("--iter", type=int)
    ap.add_argument("--output")
    args = ap.parse_args()

    plan    = yaml.safe_load(open(args.config))
    config  = plan["iterations"][args.iter]
    exp_dir = Path(f"experiments/iter_{args.iter}")
    venv    = exp_dir / "venv"

    # 创建隔离 venv（首次）
    if not (venv / "bin").exists():
        subprocess.run(["python3", "-m", "venv", str(venv)], check=True)
        subprocess.run([str(venv/"bin/pip"), "install", "-q",
                        "-r", str(exp_dir/"requirements.txt")], check=True)

    # MLflow 追踪
    mlflow.set_tracking_uri("data/mlflow")
    with mlflow.start_run(run_name=f"iter_{args.iter}") as run:
        mlflow.log_params(config)
        t0 = time.time()

        proc = subprocess.run(
            [str(venv/"bin/python"), "run.py",
             "--config", json.dumps(config)],
            cwd=str(exp_dir), capture_output=True,
            text=True, timeout=3600    # 1h CPU 超时
        )
        elapsed = time.time() - t0

        if proc.returncode != 0:
            err = summarize_error(proc.stderr)
            result = {"status": "failed", "error": err, "elapsed": elapsed}
            json.dump(result, open(args.output, "w"))
            print(json.dumps(result)); sys.exit(1)

        metrics = json.loads((exp_dir/"metrics.json").read_text())
        metrics.update({"elapsed_sec": elapsed, "mlflow_run_id": run.info.run_id})
        mlflow.log_metrics({k:v for k,v in metrics.items()
                            if isinstance(v,(int,float))})

    json.dump(metrics, open(args.output,"w"), indent=2)
    print(json.dumps(metrics, indent=2))  # Orchestrator 从 stdout 读取指标

if __name__ == "__main__": main()
